// creates list of objects from CSV file:
Cypress.Commands.add('rowsToObjects', (downloadedFileData) => {
 const csvData = downloadedFileData.split('\n').filter(e => !!e)
 const headers = csvData[0].split(';')
 const dataRows = csvData.slice(1).map(e => e.split(';'))
 const objList = dataRows.map(row => {
  return headers.reduce((prev, headerName, idx) => {
   prev[headerName] = row[idx]
   return prev
  }, {})
 })
 return objList
})
